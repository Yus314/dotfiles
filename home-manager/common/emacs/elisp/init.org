  (eval-and-compile
    (customize-set-variable
     'package-archives '(("melpa" . "https://melpa.org/packages/")
                         ("gnu" . "https://elpa.gnu.org/packages/")))

    (package-initialize)
    (unless (package-installed-p 'leaf)
  	(package-refresh-contents)
  	(package-install 'leaf))

  	:ensure t 
  	:init
  	:config 
  	(leaf-keywords-init)))
